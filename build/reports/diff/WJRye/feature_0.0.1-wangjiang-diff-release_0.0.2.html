8def get_commit_file_path_set(target_branch, current_branch, author):
9    """
10    比对 branch，获取提交的文件相对路径列表
11    :param target_branch: 要比对的分支
12    :param current_branch: 当前分支
13    :param author: git user.name
14    :return: 提交的文件相对路径列表
15    """
16    try:
17        if (target_branch == 'master' or target_branch.startswith('release')) and not current_branch.startswith(
18                'release'):
19            branch_command = f"{target_branch}..{current_branch}"
20        else:
21            branch_command = f"{current_branch}...{target_branch}"
22 
23        command = ['git', 'log', branch_command, '--author=' + author,
24                   '--name-status', '--oneline']
25        process = subprocess.Popen(command, stdout=subprocess.PIPE, stderr=subprocess.STDOUT, text=True)
26        file_path_list = set()
27        rename_file_path_list = set()
28        while True:
29            output = process.stdout.readline()
30            if output == '' and process.poll() is not None:
31                process.kill()
32                break
33            if output:
34                text = output.strip().replace("\t", "")
35                if text.startswith('M') or text.startswith('A') or text.startswith('D'):
36                    file_path = text[1:]
37                    file_path_list.add(file_path)
38                else:
39                    # 记录重命名文件，需要移除
40                    if output.strip().startswith('R'):
41                        rename_file_path = output.strip().split('\t')[1]
42                        rename_file_path_list.add(rename_file_path)
43        if len(file_path_list) == 0:
44            print(f"{' '.join(command)}: No commit files")
45            return None
46        for rename_file_path in rename_file_path_list:
47            try:
48                file_path_list.remove(rename_file_path)
49            except KeyError:
50                pass
51        return file_path_list
52    except subprocess.CalledProcessError as e:
53        print(f"Error executing command: {e}")
54        return None
